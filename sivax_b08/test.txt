create schema sivax;

SET SEARCH_PATH TO SIVAX;

CREATE TABLE PENGGUNA(
email varchar(50) PRIMARY KEY,
no_telp varchar(15) NOT NULL,
password varchar(20) NOT NULL,
status_verifikasi varchar(20) NOT NULL
);

CREATE TABLE PANITIA_PENYELENGGARA(
email varchar(50) PRIMARY KEY,
nama_lengkap varchar(30) NOT NULL,
Foreign key (email) references PENGGUNA(email) on delete cascade on update cascade
);

CREATE TABLE NAKES(
email varchar(50) PRIMARY KEY,
no_str varchar(16) NOT NULL,
tipe varchar(16) NOT NULL,
Foreign key (email) references PENGGUNA(email) on delete cascade on update cascade
);

CREATE TABLE ADMIN_SATGAS(
email varchar(50) PRIMARY KEY,
id_pegawai varchar(20) NOT NULL,
Foreign key (email) references PENGGUNA(email) on delete cascade on update cascade
);

CREATE TABLE INSTANSI(
kode varchar(20) PRIMARY KEY,
nama_instansi varchar(30) NOT NULL
);

CREATE TABLE INSTANSI_TELEPON (
kode_instansi varchar(20),
no_telp varchar(15) NOT NULL,
PRIMARY KEY (kode_instansi, no_telp),
Foreign key (kode_instansi) references INSTANSI (kode) on delete cascade on update cascade
);

CREATE TABLE INSTANSI_FASKES(
kode_instansi varchar(20) PRIMARY KEY,
tipe varchar(30) NOT NULL,	
StatusKepemilikan varchar (30) NOT NULL,
Foreign key (kode_instansi) references INSTANSI (kode) on delete cascade on update cascade
);


CREATE TABLE INSTANSI_NON_FASKES(
kode_instansi varchar(20) PRIMARY KEY,
kategori varchar(30) NOT NULL,
Foreign key (kode_instansi) references INSTANSI (kode) on delete cascade on update cascade
);

CREATE TABLE PANITIA_PADA_INSTANSI(
email_panitia varchar(50),
kode_instansi varchar(20) NOT NULL,
PRIMARY KEY (email_panitia, kode_instansi),
Foreign key (kode_instansi) references INSTANSI (kode) on delete cascade on update cascade,
Foreign key (email_panitia ) references PANITIA_PENYELENGGARA (email) on delete cascade on update cascade
);

CREATE TABLE VAKSIN(
kode varchar(20) primary key,
nama varchar(30) not null,
nama_produsen varchar(16) not null,
no_edar VARCHAR(16) unique NOT NULL,
freq_suntik INT NOT NULL,
stok INT 
);

CREATE TABLE LOKASI_VAKSIN(
kode VARCHAR(20) PRIMARY KEY,
nama VARCHAR(30) NOT NULL,
luas_ruangan INT NOT NULL,
no VARCHAR(8) NOT NULL,
jalan VARCHAR(30) NOT NULL,
kelurahan VARCHAR(30) NOT NULL,
kecamatan VARCHAR(30) NOT NULL,
kabkot VARCHAR(30) NOT NULL
);

CREATE TABLE DISTRIBUSI(
kode VARCHAR(20) PRIMARY KEY,
tanggal date NOT NULL,
biaya INT NOT NULL,
jumlah_vaksin INT NOT NULL,
kode_vaksin VARCHAR(20) NOT NULL, 
FOREIGN KEY(kode_vaksin) REFERENCES VAKSIN(kode) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE PENJADWALAN(
kode_instansi varchar(20) UNIQUE,
tanggal_waktu timestamp UNIQUE,
jumlah int not null,
kategori_penerima varchar(30) not null,
Status varchar(50) not null,
jumlah_nakes int not null,
kode_lokasi varchar(20) not null,
Email_admin varchar(50) not null,
Kode_distribusi varchar(20),
Primary key (kode_instansi, tanggal_waktu),
Foreign key (kode_instansi) references INSTANSI(kode) on delete cascade on update cascade, 
Foreign key (kode_lokasi) references LOKASI_VAKSIN(kode) on delete cascade on update cascade,
Foreign key (email_admin) references ADMIN_SATGAS(email) on delete cascade on update cascade,
Foreign key (kode_distribusi) references DISTRIBUSI(kode) on delete cascade on update cascade
);


CREATE TABLE UPDATE_STOK(
email_pegawai VARCHAR(50), 
tgl_waktu timestamp,
jumlah_update INT NOT NULL,
kode_vaksin VARCHAR(20) NOT NULL, 
PRIMARY KEY (email_pegawai,tgl_waktu),
FOREIGN KEY (email_pegawai) REFERENCES ADMIN_SATGAS(email) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (kode_vaksin) REFERENCES VAKSIN(kode) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE WARGA(
email varchar(50) PRIMARY KEY, 
nik VARCHAR(25) NOT NULL,
Nama_lengkap VARCHAR(30) NOT NULL,
jenis_kelamin VARCHAR(1) NOT NULL,
no VARCHAR(8) NOT NULL,
jalan VARCHAR(30) NOT NULL,
kelurahan VARCHAR(30) NOT NULL,
Kecamatan VARCHAR(30) NOT NULL,
Kabkot VARCHAR(30) NOT NULL,
Instansi VARCHAR(20) NOT NULL, 
FOREIGN KEY (email) REFERENCES PENGGUNA(email) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (instansi) REFERENCES INSTANSI(kode) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE STATUS_TIKET(
Kode VARCHAR(2) PRIMARY KEY,
Nama_status VARCHAR(30) NOT NULL
);

CREATE TABLE TIKET(
Email varchar(50),
No_tiket VARCHAR(10) UNIQUE,
Kode_instansi VARCHAR(20) NOT NULL, 
Kode_status VARCHAR(2) NOT NULL,
Tgl_waktu timestamp NOT NULL, 
PRIMARY KEY (email, no_tiket),
FOREIGN KEY (email) REFERENCES warga(email) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (kode_status) REFERENCES status_tiket(kode) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (kode_instansi) REFERENCES penjadwalan(kode_instansi) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY (tgl_waktu) REFERENCES penjadwalan(tanggal_waktu) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE KARTU_VAKSIN(
Email varchar(50),
No_sertifikat VARCHAR(30),
No_tiket varchar(10) not null,
Status_tahapan VARCHAR(8) NOT NULL,
PRIMARY KEY (email, no_sertifikat, no_tiket),
FOREIGN KEY (email, no_tiket) REFERENCES TIKET (email, no_tiket) ON DELETE CASCADE ON UPDATE CASCADE
);
